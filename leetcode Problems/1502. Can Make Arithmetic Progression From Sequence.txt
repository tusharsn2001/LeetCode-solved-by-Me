class Solution {
    public boolean canMakeArithmeticProgression(int[] arr) {
        int n = arr.length;
        Arrays.sort(arr);
        int b = arr[1]-arr[0];
        for(int i=1;i<n;i++)
            if(arr[i]-arr[i-1] != b)
            { 
                return false;
            }
            return true;
        
        
    }
}
